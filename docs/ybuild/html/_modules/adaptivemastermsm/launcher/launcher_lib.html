                    ref_t=300, ref_p=1.0):

    txt = """

    nsteps                  = %d
    nstlog                  = 1000

    nstxout-compressed      = 1000

    nstlist                 = 10
    ns_type                 = grid
    coulombtype             = pme
    pme_order               = 4
    fourier_spacing         = 0.16
    rvdw                    = 1.
    rcoulomb                = 1.
    tcoupl                  = v-rescale
    tc_grps                 = protein non-protein
    tau_t                   = 0.1   0.1
    ref_t                   = %d   %d
    ; Isotropic pressure coupling is now on
    Pcoupl                  = Berendsen
    Pcoupltype              = isotropic
    tau_p                   = 2.0
    ref_p                   = %f
    compressibility         = 4.5e-5
    pbc                     = xyz
    gen_vel                 = no
    """ % (dt, nsteps, constraints, ref_t, ref_t, ref_p)
    return txt
